#include<stdio.h>
#include<math.h>
#include<GL/glu.h>
#include<GL/gl.h>
#include<GL/glut.h> 
int xstart, ystart,xend,yend;
void linebres(void)
{
int dx=fabs(xend-xstart),dy=fabs(yend-ystart);
int p=2*dy-dx;
int twoDy=2*dy;
int twoDMinusDx=2*(dy-dx);
int x,y;
if(xstart>xend)
{
x=xend;
y=yend;
xend=xstart;
 }
 else
 {
x=xstart;
 }
 glBegin(GL_POINTS);
 glVertex2i(x,y);
 glEnd();
 while(x<xend)
 {
x++;
if(p<0)
p+=twoDy;
else
{
y++;
p+=twoDMinusDx;
}
glBegin(GL_POINTS);
glVertex2i(x,y);
glEnd();
 }
glFlush();
}
int main(int argc,char **argv)
{
printf("enter the value for the coordinate");
scanf("%d%d%d%d",&xstart,&xend,&ystart,&yend);
glutInit(&argc,argv);
glutInitDisplayMode(GLUT_SINGLE|GLUT_RGB);
glutInitWindowPosition(50,100);
glutInitWindowSize(500,500);
glutCreateWindow("An example program");
glClearColor(1.0,1.0,1.0,0.0);
glMatrixMode(GL_PROJECTION);
gluOrtho2D(0.0,200.0,0.0,150.0);
glClear(GL_COLOR_BUFFER_BIT);
glColor3f(1.0,0.0,0.0);
glPointSize(5.0);
glutDisplayFunc(linebres);
glutMainLoop();
 }
